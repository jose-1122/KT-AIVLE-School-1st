###크리스마스 트리
dict_x = {111:'green', 148:'lightgreen', 185:'yellowgreen',
          222:'green', 259:'lightgreen', 296:'yellowgreen',
          333:'green', 370:'lightgreen', 407:'yellowgreen',
          444:'green', 481:'lightgreen', 518:'yellowgreen',
          555:'green', 592:'lightgreen', 629:'gold',
          666:'green', 703:'lightgreen', 740:'brown',
          777:'green'}
for k in dict_x.keys():
    X = [np.cos(i/2)*(k-i) for i in range(k)]
    Y = [np.sin(i/9)*(k-i) for i in range(k)]
    Z = [i for i in range(k)]
    ax.scatter(X,Y,Z, s=50, c=dict_x[k], marker='.')
